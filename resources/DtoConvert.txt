
import com.google.gson.Gson;

import java.lang.reflect.ParameterizedType;
import java.lang.reflect.Type;
import java.util.List;

public class DtoConvert {
    private static Gson gson = new Gson();

    private DtoConvert() {
    }

    public static <T> T convert(Object val, Class<T> clazz) {
        return gson.fromJson(gson.toJson(val), clazz);
    }

    public static <T> List<T> convert2List(List valList , final Class<T> clazz){
        ParameterizedType parameterizedType = new ParameterizedType() {
            @Override
            public Type[] getActualTypeArguments() {
                return new Type[]{clazz};
            }

            @Override
            public Type getRawType() {
                return List.class;
            }

            @Override
            public Type getOwnerType() {
                return null;
            }
        };
        return gson.fromJson(gson.toJson(valList) , parameterizedType);
    }

}
